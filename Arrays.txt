#include<iostream>
using namespace std;

int arr[50];
int size;

void CreateArray(){
	cout<<"\nCreating an Array";
   	 	 cout<<"\nEnter the number of the elements = ";
   	 	 cin>>size;
   	 	  int i;
			  for(i=0;i<size;i++){
			  	cout<<"\nElement "<<i+1<<" = ";
                cin>>arr[i];
			  }
}

void DisplayArray(){
	int i;
	cout<<"\nDisplaying the Array\n";		  
			  for(i=0;i<size;i++){
			  	cout<<arr[i]<<"\n";
			  }
}

void InsertArray(){
	int i;
	int num,pos;
	 cout<<"\nInserting into the Array";
		  cout<<"\nEnter the element =";
		  cin>>num;
		  cout<<"\nEnter the position =";
		  cin>>pos;
		  
		  for(i=size;i>=pos-1;i--){
		  	arr[i]=arr[i-1];
		  }
		  arr[pos-1]=num;
		  size++;
}

void DeleteArray(){
	int pos;
	cout<<"Enter the position to delete: ";
	cin>>pos;
	int i;
	for(i=pos-1;i<size;i++){
		arr[i]=arr[i+1];
	}
	size--;
}

void UpdateArray(){
	int num,pos;
	cout<<"\nUpdating the Array";
		 cout<<"\nEnter the element =";
		  cin>>num;
		  cout<<"\nEnter the position =";
		  cin>>pos;
		  arr[pos-1]=num;
}

int main(){
	int choice;
	char opt;
	do{
		cout<<"\n1. Creating the Array";
		cout<<"\n2. Displaying the Array";
		cout<<"\n3. Inserting in the Array";
		cout<<"\n4. Delete the Array";
        cout<<"\n5. Update the Array"; 
		  cout<<"\nEnter the choice = ";
		  cin>>choice;
		  
		  switch(choice){
		  	case 1:
		  		
   	          CreateArray();
			  break;
			  
	 case 2:
	 	
	       DisplayArray();
			  break;
    
	 case 3:
	      
		InsertArray();
        break;
        
     case 4:
     	
	    DeleteArray();
		  break;
		  
    case 5:
    	
    	UpdateArray();
    	break;
		    
   }
    
	cout<<"\nDo you want to continue (Y/N) = ";
    cin>>opt;
   
   }while(opt=='Y'|| opt=='y');
   cout<<"THANK YOU FOR COMING !";
	return 0;	  
}
