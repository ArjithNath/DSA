#include<iostream>
using namespace std;

void BubbleSort(int arr[], int n){
	int i;
	int j;
	
	for(i=0;i<n-1;i++){
		//Each pass
		for(j=0;j<n-i-1;j++){
			
			if(arr[j]>arr[j+1]){
				//If the current element is greater than next, swap the variables
				int temp=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
				                      
				
			}
		}
	}
}

void PrintArray(int arr[], int n){
	int i;
	
	for(i=0;i<n;i++){
		
		cout<<arr[i]<<endl;
		
	}
	
}
int main(){
	int arr[50],n;
	int i;
	
	cout<<"Enter the number of elements: ";
	cin>>n;
	cout<<"\nEnter "<<n<<" Elements:"<<endl;
	
	for(i=0;i<n;i++){
		
		cin>>arr[i];
		
	}
	
	cout<<"\nOriginal Array: "<<endl;
	PrintArray(arr,n);
	BubbleSort(arr,n);
	cout<<"\nSorted Array: "<<endl;
	PrintArray(arr,n);
	
	return 0;
}
